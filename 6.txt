using System;
using System.Web.UI;

namespace WebApplication5
{
    public partial class ForLoopDropDown : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                string[] languages = { "C", "C++", "Java", "Python", "ASP" };

                for (int i = 0; i < languages.Length; i++)
                {
                    ddlLanguages.Items.Add(languages[i]);
                }
            }
        }

        protected void btnSubmit_Click(object sender, EventArgs e)
        {
            lblSelected.Text = "You selected: " + ddlLanguages.SelectedItem.Text;
        }
    }
}
