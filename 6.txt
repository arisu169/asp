using System;
using System.Web.UI;
namespace WebApplication21
{
    public partial class ForLoop : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                for (int i = 1; i <= 5; i++)
                    ddlNumbers.Items.Add(i.ToString());
            }}}}
