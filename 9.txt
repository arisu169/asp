using System;
using System.Data.SqlClient;
public partial class SelectCreate : System.Web.UI.Page {
protected void Page_Load(object sender, EventArgs e) {
var c=new SqlConnection("Data Source=.;Initial Catalog=master;Integrated
Security=True");
c.Open();
new SqlCommand("if object_id('t') is null create table t(id int)",c).ExecuteNonQuery();
var r=new SqlCommand("select * from t",c).ExecuteReader();
lblResult.Text=r.HasRows?"Rows Found":"No Rows";
r.Close(); c.Close();
}
}
