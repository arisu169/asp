using System;
using System.Web.UI;

namespace WebApplication4
{
    public partial class StudentForm : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
        }

        protected void btnSubmit_Click(object sender, EventArgs e)
        {
            string name = txtName.Text;
            string roll = txtRoll.Text;
            string studentClass = ddlClass.SelectedItem.Text;
            string status = rblStatus.SelectedItem.Text;

            lblMessage.Text = $"Student {name} (Roll: {roll}) of {studentClass} is marked {status}.";
        }
    }
}
