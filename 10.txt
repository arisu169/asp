using System;
using System.Data.SqlClient;
public partial class InsertUpdate : System.Web.UI.Page {
protected void Page_Load(object sender, EventArgs e) {
var c=new SqlConnection("Data Source=.;Initial Catalog=master;Integrated
Security=True");
c.Open();
new SqlCommand("if not exists(select * from t where id=1) insert into t
values(1)",c).ExecuteNonQuery();
new SqlCommand("update t set id=2 where id=1",c).ExecuteNonQuery();
lblAction.Text="Inserted and Updated";
c.Close();
}
}
