using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace WebApplication7
{
    public partial class WhileLoopMenu : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            int i = 1;
            while (i <= 6) // Add 6 items
            {
                MenuItem item = new MenuItem("BCA " + i + " Semester");
                m1.Items.Add(item);
                i++;
            }
        }
    }
}
