using System;
using System.Web.UI;
using System.Web.UI.WebControls;
namespace WebApplication21
{
    public partial class WhileLoop : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                int i = 1;
                while (i <= 1)
                    menu.Items.Add(new MenuItem("Item" + i++));
            }}}}
