using System;
using System.Data.SqlClient;
using System.Web.UI;
namespace WebApplication22
{
    public partial class DataSource: Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            var c = new SqlConnection(@"Data Source=COLLEGE_SERVER_IP_OR_NAME;Initial Catalog=COLLEGE_DATABASE_NAME;User ID=YOUR_USERNAME;Password=YOUR_PASSWORD");
            c.Open();
            lblStatus.Text = "Connected";
            c.Close();
        }
    }
}
